// src/router/index.js 底部添加
router.beforeEach((to, from, next) => {
  const token = localStorage.getItem('token')
  
  // 需要认证的路由
  const requiresAuth = to.matched.some(record => 
    record.component?.name === 'MainLayout'
  )
  
  if (requiresAuth && !token) {
    next('/auth/login')
  } else if (to.path === '/auth/login' && token) {
    next('/dashboard/console')
  } else {
    next()
  }
})

export default router

#### 4. Tabs标签页刷新后丢失
**位置**: `src/stores/tabs.js`

- 页面刷新后tabsList数组被清空
- 没有持久化存储机制
- activeTab状态也丢失

**解决方案**:

// src/stores/tabs.js - 添加持久化
import { defineStore } from 'pinia'
import { ref, watch } from 'vue'
import { useRouter } from 'vue-router'

export const useTabsStore = defineStore('tabs', () => {
  const router = useRouter()
  
  // 从localStorage恢复数据
  const savedTabs = localStorage.getItem('tabsList')
  const savedActive = localStorage.getItem('activeTab')
  
  const tabsList = ref(savedTabs ? JSON.parse(savedTabs) : [])
  const activeTab = ref(savedActive || '')

  // 监听变化并保存
  watch(tabsList, (newVal) => {
    localStorage.setItem('tabsList', JSON.stringify(newVal))
  }, { deep: true })
  
  watch(activeTab, (newVal) => {
    localStorage.setItem('activeTab', newVal)
  })

})

#### 5. 主题切换在index.html中未初始化
**位置**: `src/main.js` (第22-33行) + `index.html`
**问题**: 
- 虽然在main.js中添加了防闪烁逻辑,但此时DOM已经渲染
- 在慢速设备或网络下仍可能看到主题闪烁
**影响**: 暗黑模式切换时可能出现短暂白屏闪烁
**解决方案**: 
```html
<!-- index.html 的 <head> 中添加 -->
<head>
  <!-- ... existing code ... -->
  <script>
    // 在HTML解析阶段立即执行,避免闪烁
    ;(function() {
      const savedMode = localStorage.getItem('themeMode') || 'auto'
      let isDark = false
      
      if (savedMode === 'dark') {
        isDark = true
      } else if (savedMode === 'auto') {
        isDark = window.matchMedia('(prefers-color-scheme: dark)').matches
      }
      
      if (isDark) {
        document.documentElement.classList.add('dark')
      }
    })()
  </script>
</head>
```

#### 6. 请求拦截器缺少Token注入
**位置**: `src/utils/request.js`
**问题**: 
- 只有响应拦截器,没有请求拦截器
- Token存储在localStorage但未自动添加到请求头
- 每个API调用都需要手动添加headers
**影响**: 后端无法识别用户身份,需要额外代码处理
**解决方案**: 
```javascript
// src/utils/request.js - 添加请求拦截器
service.interceptors.request.use(
  config => {
    const token = localStorage.getItem('token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  error => {
    return Promise.reject(error)
  }
)

// ... existing response interceptor code ...
```

#### 7. Vite缺少API代理配置
**位置**: `vite.config.js`
**问题**: 
- 开发环境请求/api会直接发送到当前域名
- 没有配置proxy转发到后端服务器
- 生产环境部署时会出现跨域问题
**影响**: 前后端分离部署时无法联调
**解决方案**: 
```javascript
// vite.config.js
export default defineConfig({
  plugins: [vue()],
  base: '/myvue3/',
  server: {
    proxy: {
      '/api': {
        target: 'http://localhost:3000', // 后端地址
        changeOrigin: true,
        rewrite: (path) => path.replace(/^\/api/, '')
      }
    }
  },
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url))
    },
  },
})
```

- ✅ 组件拆分合理,职责明确
