# Vue3 核心技术实现详解

## 1. 动态路由与权限控制

### 核心概念

**权限控制流程：**
```
用户登录 → 获取权限列表 → 存储到Pinia → 路由守卫拦截 → 过滤路由 → 动态注册 → 菜单渲染
```

### 1.1 Pinia Store 管理权限

```javascript
// stores/permission.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { asyncRoutes } from '@/router/async'

export const usePermissionStore = defineStore('permission', () => {
  // 存储用户权限码列表（如 ['dashboard:view', 'user:add']）
  const permissions = ref([])
  
  // 存储过滤后的可访问路由
  const accessibleRoutes = ref([])
  
  // 计算属性：判断是否有某权限
  const hasPermission = computed(() => {
    return (permissionCode) => {
      return permissions.value.includes(permissionCode)
    }
  })
  
  // 设置权限列表
  function setPermissions(permissionList) {
    permissions.value = permissionList
  }
  
  // 递归过滤路由（核心方法）
  function filterRoutes(routes) {
    const result = []
    
    routes.forEach(route => {
      // 深拷贝路由对象（保留component函数）
      const routeCopy = { ...route }
      
      // 如果路由没有meta或不需要权限，直接添加
      if (!routeCopy.meta || !routeCopy.meta.permissions) {
        result.push(routeCopy)
        return
      }
      
      // 检查当前用户是否有权限访问此路由
      const hasAccess = routeCopy.meta.permissions.some(permission => 
        permissions.value.includes(permission)
      )
      
      if (hasAccess) {
        // 如果有子路由，递归过滤
        if (routeCopy.children && routeCopy.children.length > 0) {
          routeCopy.children = filterRoutes(routeCopy.children)
        }
        result.push(routeCopy)
      }
    })
    
    return result
  }
  
  // 生成可访问的路由
  function generateRoutes() {
    const filtered = filterRoutes(asyncRoutes)
    accessibleRoutes.value = filtered
    return filtered
  }
  
  return {
    permissions,
    accessibleRoutes,
    hasPermission,
    setPermissions,
    generateRoutes
  }
})
```

### 1.2 路由守卫中动态注册路由

```javascript
// router/index.js
import { createRouter, createWebHistory } from 'vue-router'
import { constantRoutes } from './constant'
import { usePermissionStore } from '@/stores/permission'
import { useUserStore } from '@/stores/user'

const router = createRouter({
  history: createWebHistory(),
  routes: constantRoutes // 先注册常量路由
})

// 标记是否已添加动态路由
let hasAddedRoutes = false

// 路由守卫
router.beforeEach(async (to, from, next) => {
  const userStore = useUserStore()
  const permissionStore = usePermissionStore()
  
  // 判断是否已登录
  if (userStore.token) {
    if (to.path === '/login') {
      // 已登录，跳转到首页
      next({ path: '/' })
    } else {
      // 判断是否已添加动态路由
      if (hasAddedRoutes) {
        next()
      } else {
        try {
          // 1. 从后端获取用户信息和权限列表
          const userInfo = await userStore.getUserInfo()
          
          // 2. 设置权限到Store
          permissionStore.setPermissions(userInfo.permissions)
          
          // 3. 过滤并生成可访问的路由
          const accessRoutes = permissionStore.generateRoutes()
          
          // 4. 动态注册路由（核心：使用addRoute）
          accessRoutes.forEach(route => {
            router.addRoute(route)
          })
          
          // 5. 标记已添加
          hasAddedRoutes = true
          
          // 6. 重要：重新导航到目标路由（因为刚添加的路由需要重新匹配）
          next({ ...to, replace: true })
        } catch (error) {
          // 出错则重置并跳转登录页
          userStore.resetToken()
          hasAddedRoutes = false
          next('/login')
        }
      }
    }
  } else {
    // 未登录，判断是否在白名单中
    if (['/login', '/register'].includes(to.path)) {
      next()
    } else {
      next('/login')
    }
  }
})

export default router
```

## 2. 大数据量表格优化 - 虚拟滚动

### 核心原理

**传统表格的问题：**
- 10000条数据 = 10000个DOM节点
- 内存占用高、渲染慢、滚动卡顿

**虚拟滚动的解决方案：**
- 只渲染可视区域的约20-30个节点
- 滚动时动态替换数据，不创建新DOM
- 固定行高 + 绝对定位

### 2.1 安装依赖

```bash
npm install vue-virtual-scroller
```

```vue
  pageSize.value = newSize
  currentPage.value = 1
  fetchData()

  currentPage.value = newPage
  fetchData()
}

function handlePageSizeChange() {
  currentPage.value = 1
  fetchData()
}

// ========== 编辑和删除 ==========
function handleEdit(user) {
  console.log('编辑用户:', user)
  // 打开编辑对话框
}

      type: 'warning'

    fetchData()

      ElMessage.error('删除失败')

  }

onMounted(() => {
```

### 2.3 关键参数详解

#### RecycleScroller 组件参数

| 参数 | 类型 | 说明 | 示例值 |
|------|------|------|--------|
| `items` | Array | 要渲染的数据数组 | `[...]` |
| `item-size` | Number | **固定行高（像素）**，必须设置 | `50` |
| `key-field` | String | 唯一标识字段名 | `"id"` |
| `v-slot="{ item }"` | Slot | 作用域插槽，接收当前项 | - |

#### 为什么需要固定行高？

```javascript
// 虚拟滚动内部计算逻辑（简化版）
const visibleStartIndex = Math.floor(scrollTop / itemSize)
const visibleEndIndex = Math.ceil((scrollTop + containerHeight) / itemSize)
const visibleItems = items.slice(visibleStartIndex, visibleEndIndex)

// 通过绝对定位模拟位置
const itemStyle = {
  position: 'absolute',
  top: index * itemSize + 'px', // 根据索引计算位置
  height: itemSize + 'px',
  width: '100%'
}
```

**如果不固定行高会怎样？**
- ❌ 无法准确计算滚动位置
- ❌ 元素会重叠或错位
- ✅ 如需可变高度，使用 `DynamicScroller`（性能略差）

### 2.4 性能对比

| 场景 | 传统el-table | RecycleScroller |
|------|-------------|-----------------|
| 100条数据 | DOM节点: 100 | DOM节点: ~5 |
| 10000条数据 | DOM节点: 10000 | DOM节点: ~30 |
| 内存占用 | ~50MB | ~2MB |
| 首次渲染 | ~2秒 | ~100ms |
| 滚动FPS | 10-20帧 | 60帧 |

## 3. 表单组件封装 - BaseForm

### 3.1 组件实现

```vue
<!-- components/common/BaseForm.vue -->
<template>
  <el-form
    ref="formRef"
    :model="formData"
    :rules="computedRules"
    :label-width="labelWidth"
  >
    <el-row :gutter="20">
      <el-col
        v-for="schema in schemas"
        :key="schema.prop"
        :span="schema.span || 12"
      >
        <el-form-item :label="schema.label" :prop="schema.prop">
          <!-- 输入框 -->
          <el-input
            v-if="schema.type === 'input'"
            v-model="formData[schema.prop]"
            :placeholder="schema.placeholder"
            :disabled="schema.disabled"
          />
          
          <!-- 密码框 -->
          <el-input
            v-else-if="schema.type === 'password'"
            v-model="formData[schema.prop]"
            type="password"
            show-password
            :placeholder="schema.placeholder"
          />
          
          <!-- 下拉选择 -->
          <el-select
            v-else-if="schema.type === 'select'"
            v-model="formData[schema.prop]"
            :placeholder="schema.placeholder"
            :multiple="schema.multiple"
            style="width: 100%"
          >
            <el-option
              v-for="option in schema.options"
              :key="option.value"
              :label="option.label"
              :value="option.value"
            />
          </el-select>
          
          <!-- 开关 -->
          <el-switch
            v-else-if="schema.type === 'switch'"
            v-model="formData[schema.prop]"
          />
          
          <!-- 日期选择器 -->
          <el-date-picker
            v-else-if="schema.type === 'date'"
            v-model="formData[schema.prop]"
            type="date"
            :placeholder="schema.placeholder"
            style="width: 100%"
          />
        </el-form-item>
      </el-col>
    </el-row>
    
    <!-- 按钮区域 -->
    <el-form-item>
      <el-button type="primary" @click="handleSubmit">
        {{ submitText }}
      </el-button>
      <el-button @click="handleReset">重置</el-button>
      <el-button @click="$emit('cancel')">取消</el-button>
    </el-form-item>
  </el-form>
</template>

<script setup>
import { ref, reactive, computed, watch } from 'vue'

// ========== Props定义 ==========
const props = defineProps({
  // 表单配置数组
  schemas: {
    type: Array,
    required: true,
    default: () => []
  },
  // 初始数据
  modelValue: {
    type: Object,
    required: true,
    default: () => ({})
  },
  // 标签宽度
  labelWidth: {
    type: String,
    default: '100px'
  },
  // 提交按钮文字
  submitText: {
    type: String,
    default: '提交'
  },
  // 表单模式：add(新增) 或 edit(编辑)
  mode: {
    type: String,
    default: 'add',
    validator: (val) => ['add', 'edit'].includes(val)
  }
})

// ========== Emits定义 ==========
const emit = defineEmits(['update:modelValue', 'submit', 'cancel'])

// ========== 响应式数据 ==========
const formRef = ref(null)
const formData = reactive({ ...props.modelValue })

// 监听外部数据变化
watch(() => props.modelValue, (newVal) => {
  Object.assign(formData, newVal)
}, { deep: true })

// ========== 动态校验规则（核心）==========
const computedRules = computed(() => {
  const rules = {}
  
  props.schemas.forEach(schema => {
    const ruleList = []
    
    // 必填校验
    if (schema.required) {
      // 根据模式动态调整（如编辑时密码可选）
      const isRequired = typeof schema.required === 'function'
        ? schema.required(props.mode)
        : schema.required
      
      if (isRequired) {
        ruleList.push({
          required: true,
          message: `${schema.label}不能为空`,
          trigger: 'blur'
        })
      }
    }
    
    // 自定义校验
    if (schema.validator) {
      ruleList.push({
        validator: schema.validator,
        trigger: 'blur'
      })
    }
    
    // 长度校验
    if (schema.min || schema.max) {
      ruleList.push({
        min: schema.min,
        max: schema.max,
        message: `长度在${schema.min}到${schema.max}个字符`,
        trigger: 'blur'
      })
    }
    
    rules[schema.prop] = ruleList
  })
  
  return rules
})

// ========== 表单操作方法 ==========
async function handleSubmit() {
  try {
    await formRef.value.validate()
    emit('submit', { ...formData })
  } catch (error) {
    console.error('表单验证失败:', error)
  }
}

function handleReset() {
  formRef.value.resetFields()
}

// 暴露方法给父组件
defineExpose({
  validate: () => formRef.value.validate(),
  resetFields: () => formRef.value.resetFields(),
  clearValidate: () => formRef.value.clearValidate()
})
</script>
```

```vue

async function handleSubmit(data) {

    if (formMode.value === 'add') {
      await createUser(data)
      ElMessage.success('创建成功')
    } else {
      await updateUser(data)
      ElMessage.success('更新成功')
    }
    dialogVisible.value = false
    fetchData() // 刷新列表
  } catch (error) {
    ElMessage.error('操作失败')
  }
}
</script>
```

## 5. RBAC 权限模型

### 5.1 三级关联模型

```
用户(User) ──多对多──> 角色(Role) ──多对多──> 权限(Permission)
```

```javascript

  id: 1,
  name: "管理员",
```

## 6. 路由懒加载优化

### 6.1 什么是路由懒加载？

**传统导入（立即加载）：**
```javascript
import UserView from '@/views/system/user.vue'
// ❌ 页面初始化时就加载，增加首屏时间
```

**懒加载（按需加载）：**
```javascript
const UserView = () => import('@/views/system/user.vue')
// ✅ 只有访问该路由时才加载
```

### 6.2 实现原理

```javascript
// webpack/vite 会将每个 import() 分割成独立的 chunk（代码块）
// 浏览器只在需要时下载对应的 chunk

// 打包结果示例：
// dist/
//   ├── index.js          (主包)
//   ├── user-abc123.js    (用户管理页面)
//   ├── role-def456.js    (角色管理页面)
//   └── dashboard-ghi789.js (仪表盘页面)
```

### 6.3 实际应用

```javascript
// router/async.js

// ✅ 推荐：所有异步组件都使用懒加载
const Layout = () => import('@/components/layout/MainLayout.vue')

export const asyncRoutes = [
  {
    path: '/dashboard',
    component: Layout,
    children: [
      {
        path: 'console',
        component: () => import('@/views/dashboard/console.vue'), // 懒加载
        meta: { title: '控制台' }
      }
    ]
  }
]

// ❌ 避免：静态导入会导致包体积过大
// import Console from '@/views/dashboard/console.vue'
```

### 6.4 性能提升效果

| 指标 | 静态导入 | 懒加载 |
|------|---------|--------|
| 首屏JS体积 | 2.5MB | 800KB |
| 首屏加载时间 | 3.2秒 | 1.1秒 |
| 按需加载单个页面 | - | 200-400ms |
